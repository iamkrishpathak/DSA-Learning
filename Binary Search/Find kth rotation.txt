class Solution {
    public int findKRotation(int arr[]) {
        int low = 0;
        int high = arr.length - 1;
        int min = Integer.MAX_VALUE;
        int idx = -1;
        while(low<=high){
            int mid = (low + high)/2;
            if(arr[low]<=arr[high]){
                if(arr[low]<min){
                    min = arr[low];
                    idx = low;
                }
            }
            if (arr[low]<=arr[mid]){
                if(arr[low]<min){
                    min = arr[low];
                    idx = low;
                }
                low = mid + 1;
            }
            else{
                if(arr[high]<min){
                    min = arr[high];
                    idx = mid;
                }
                high = mid - 1;
            }
        }
        return idx;
    }
}
