class Solution {
    int countFreq(int[] arr, int target) {
        // code here
        int low = 0, high = arr.length - 1;
        int lowocc = -1;
        int highocc = -1;
        while(low<=high){
            int mid = (low+high)/2;
            if(arr[mid] == target){
               lowocc = mid;
               high = mid - 1;
            }
            else if(arr[mid] < target){
                low = mid + 1;
            }
            else{
                high = mid - 1;
            }
        }
        low = 0;
        high = arr.length - 1;
        while(low<=high){
            int mid = (low+high)/2;
            if(arr[mid] == target){
               highocc = mid;
               low = mid + 1;
            }
            else if(arr[mid] < target){
                low = mid + 1;
            }
            else{
                high = mid - 1;
            }
        }
        if(lowocc == -1){
            return 0;
        }
        else{
            return (highocc - lowocc) + 1;
        }
    }
}


It is similar to First and Last Occurence just here we are return the number of occurences rather than the indices of both occurences
